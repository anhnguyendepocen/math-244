Ridge regression

X <- model.matrix(Balance ~ ., data = credit)[, -1]
X[1:2, ]
##    Income Limit Rating Cards Age Education GenderFemale StudentYes
## 1  14.891  3606    283     2  34        11            0          0

##   MarriedYes EthnicityAsian EthnicityCaucasian

Y <- credit$Balance
lambdas <- seq(from = 1e4, to = 1e-2, length.out = 100)
library(glmnet)
rm1 <- glmnet(x = X, y = Y, alpha = 0, lambda = lambdas, standardize = TRUE)
